private async Task<int> GetMaxIdNo(int itemMasterID, int bookingID)
{
    var query = $@"Select ISNULL(MAX(RollNo), 0) MaxRollNo
        From tbl
        Where ItemMasterID = {itemMasterID} And BookingID = {bookingID}";
    return await _service.GetSingleIntFieldAsync(query);
}
